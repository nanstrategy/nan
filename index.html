function saveRow(d) {

// doPost: รับจาก fetch (JSON body)
function doPost(e) {
  try {
    const raw = (e.postData && e.postData.contents) ? e.postData.contents : '{}';
    const d = JSON.parse(raw);
    if (d._ping) return ContentService.createTextOutput('ok').setMimeType(ContentService.MimeType.TEXT);
    const row = saveRow(d);
    return ContentService.createTextOutput(JSON.stringify({status:'ok',rows:row})).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error',msg:err.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

// doGet: รับทั้ง form submit (?payload=...) และ dashboard (?callback=...)
function doGet(e) {
  try {
    const param = e && e.parameter ? e.parameter : {};

    // กรณีส่งข้อมูลจาก hidden form (?payload=JSON)
    if (param.payload) {
      const d = JSON.parse(param.payload);
      if (!d._ping) {
        const row = saveRow(d);
        // ต้องคืน HTML เพราะ iframe จะ load ผลลัพธ์
        return HtmlService.createHtmlOutput('<script>if(window.parent)window.parent.postMessage("ok","*");</script>ok');
      }
      return ContentService.createTextOutput('ok').setMimeType(ContentService.MimeType.TEXT);
    }

    // กรณีดึงข้อมูล dashboard (?callback=fnName)
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName('nan_survey');
    let rows = [];
    if (sheet && sheet.getLastRow() > 1) {
      const data = sheet.getDataRange().getValues();
      const keys = data[0];
      rows = data.slice(1).map(r => {
        const obj = {};
        keys.forEach((k,i) => obj[k] = r[i]);
        return obj;
      });
    }
    const json = JSON.stringify(rows);
    const cb = param.callback;
    if (cb) {
      return ContentService.createTextOutput(cb + '(' + json + ')').setMimeType(ContentService.MimeType.JAVASCRIPT);
    }
    return ContentService.createTextOutput(json).setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    const cb = e && e.parameter && e.parameter.callback;
    const errJson = JSON.stringify({status:'error',msg:err.toString()});
    if (cb) return ContentService.createTextOutput(cb+'('+errJson+')').setMimeType(ContentService.MimeType.JAVASCRIPT);
    return ContentService.createTextOutput(errJson).setMimeType(ContentService.MimeType.JSON);
  }
}

🚀 ขั้นตอนที่ 2 — Deploy เป็น Web App (สำคัญมาก)

  1. คลิก Deploy → New deployment
  2. กดไอคอน ⚙️ เลือก Web app
  3. ตั้งค่าให้ครบ:
      • Execute as: Me (your email)
      • Who has access: Anyone ⚠️ ต้องเลือก Anyone เท่านั้น
  4. กด Deploy → อนุญาตสิทธิ์ทุกอย่างที่ขอ (Allow All)
  5. คัดลอก Web app URL ใส่ในช่องด้านล่าง
⚠️ ถ้าแก้ไขโค้ดแล้ว ต้อง Deploy ใหม่ทุกครั้ง โดยคลิก Deploy → Manage deployments → แก้ไข (ดินสอ) → Version: New version → Deploy
